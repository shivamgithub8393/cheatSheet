  3. fetch API github prototype with link-

    
//--------------------------------------------------------------------------------------------------
// fetch API prototype
//--------------------------------------------------------------------------------------------------
fetch('url')  // Default request method is GET
  .then(res => res.json())
  .then(data => console.log(data))
  .catch(error => console.log(error));


// request prototype with POST method
fetch('url', {
  method: "POST",
  body: JSON.stringify(data),
  headers: {
    "Content-Type": "application/json"
  },
  credentials: "same-origin"
})
  .then(function(response) {
    response.status     //=> number 100â€“599
    response.statusText //=> String
    response.headers    //=> Headers
    response.url        //=> String

    return response.text()
  }, function(error) {
    error.message //=> String
  })

// link to github documentation- https://github.github.io/fetch/

  4. Create Database and Table Schema Using Pymysql in Python 

    
import pymysql

### create connection to mysql server

'localhost',

## Finally execute the query
cur.execute(sql)
