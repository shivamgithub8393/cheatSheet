  3. React JS

    
//--------------------------------------------------------------------------------------------------
// Create Routes using "react-router-dom"
//--------------------------------------------------------------------------------------------------
import { BrowserRouter as Router, Route, Switch, Link } from 'react-router-dom'

function App() {
return (
<Router>
  <div>
    <nav>
      <ul>
        <li>
          <Link to="/">Home</Link>
        </li>
        <li>
          <Link to="/about">About</Link>
        </li>
        <li>
          <Link to="/users">Users</Link>
        </li>
      </ul>
    </nav>
    {/* A <Switch> looks through its children <Route>s and
                    renders the first one that matches the current URL. */}
    <Switch>
      <Route exact path="/"> {/* exact is used for check exactly same path */}
        <Home />
      </Route>
      <Route path="/about">
        <About />
      </Route>
      <Route path="/users">
        <Users />
      </Route>
    </Switch>
  </div>
</Router>
);
}
const Home = () => {
  return <h2>Home</h2>
}
const About = () => {
  return <h2>About</h2>
}
const Users = () => {
  return <h2>Users</h2>
}
            
export default App;

### create connection to mysql server
conn = pymysql.connect(
'localhost',
'root',
'Shivam123',

)

## Create cursor

sql = "CREATE TABLE IF NOT EXISTS pythonlogin.accounts ( Id INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY,name VARCHAR(300)
NOT NULL,password VARCHAR(3000) NOT NULL,email VARCHAR(150)) "

## Finally execute the query
cur.execute(sql)
